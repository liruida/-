架构和算法不是同一个东西，目标不同，分而思之。虽然他们存在一定关系，例如什么样的算法适用于什么样的架构，架构的合理选择有利于算法的实现。但算法的得来在思维，不在架构，混而思之，徒加难度。

实现太阳运行位置与经纬度对应关系
#include <iostream>
using namespace std;
int main()
{
    double d=23.5/90;
    double w=0.0;
    for(int j=1;j<5;j++){
           for(int i=1;i<91;i++){
                if(j==1){   
                w=d*(i-1);   
                cout<<"季度"<<j<<"（春分至夏至）第"<<i<<"天"<<"北纬：" <<w<<"度"<< endl;
                  
                  }
              
               } 
                system("pause");
              for(int i=1;i<91;i++){
               if(j==2){   
                w=23.5-d*(i-1);   
                cout<<"季度"<<j<<"（夏至至秋分）第"<<i<<"天"<<"北纬：" <<w<<"度"<< endl;
                 
                  }
                    
                  if(j==3){   
                w=d*(i-1);   
                cout<<"季度"<<j<<"（秋分至冬至）第"<<i<<"天"<<"南纬：" <<w<<"度"<< endl;
                  
                  }
              
              if(j==4){   
                w=23.5-d*(i-1);   
                cout<<"季度"<<j<<"（冬至至春分）第"<<i<<"天"<<"南纬：" <<w<<"度"<< endl;
                
                }
                     
                         
                          }
            }
           system("pause");

    return 0;
}

五行交互由气也，木星之大，引得小行星之围绕粉碎，护卫地球，如人之肝，清毒。 太阳之风暴与金地共线时使金气层之离子入地球，旱疫也。

天地人同气。地之济水流动若火星之轨迹，三入地三出地。火星二小卫星，力弱，火星公转倾斜度0~50度之间，故人心易扰，须修心也。

物，气之现，非物理之物；气之所受，地利天时人和。木之活木，受土水火之气；死木则僵。河中之木耐湿；沙中之木耐旱，易河则死；此之谓也。
